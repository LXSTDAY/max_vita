void main(
  float3 Position,
  float2 TexCoord0,
  float4 Color0,
  uniform float4x4 WorldToView,
  uniform float4x4 ObjToWorld,
  half2 out Out_Tex0 : TEXCOORD0,
  fixed4 out Out_Color0 : COLOR0,
  float4 out gl_Position : POSITION
) {
  float4 WorldPos = mul(float4(Position, 1.0), transpose(ObjToWorld));
  gl_Position = mul(transpose(WorldToView), WorldPos);
  Out_Tex0 = TexCoord0 * 0.00390625; // 1 / 256.0
  Out_Color0 = Color0.bgra;
}
