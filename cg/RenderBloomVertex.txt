void main(
  float3 Position,
  float2 TexCoord0,
  uniform half3 ColorMult,
  half2 out Out_Tex0 : TEXCOORD0,
  fixed out alpha : COLOR0,
  float4 out gl_Position : POSITION
) {
  gl_Position =  float4(Position.x, -Position.y, Position.z, 1.0);
  Out_Tex0 = (TexCoord0 * 0.00390625) + ColorMult.xy; // 1 / 256.0
  alpha = ColorMult.z;
}
